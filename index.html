// BANCO DE DADOS REGIONAL DE CULTURAS — BUSCABOANERGES const BANCO_REGIONAL = { "sp_interior_central": { dominantes: ["Cana-de-açúcar", "Milho Safrinha"], complementares: ["Uva", "Mandioca", "Hortaliças"], proibidas: ["Soja verão", "Algodão"], bioma: "Mata Atlântica", clima: "Tropical de altitude" }, "sp_noroeste": { dominantes: ["Cana-de-açúcar", "Soja", "Milho"], complementares: ["Laranja", "Pastagem"], proibidas: ["Arroz de várzea", "Culturas de serra"], bioma: "Cerrado", clima: "Tropical" } // Demais regiões seguem a mesma estrutura }; // FUNÇÃO DE VALIDAÇÃO — NÃO PERMITE INDICAÇÃO INCOERENTE function validarCultura(regiao, cultura, clima) { const dadosRegiao = BANCO_REGIONAL[regiao]; const ehForte = dadosRegiao.dominantes.includes(cultura); const ehComplementar = dadosRegiao.complementares.includes(cultura); const ehProibida = dadosRegiao.proibidas.includes(cultura); if (ehProibida) return { status: "INCOMPATÍVEL", mensagem: "Cultura não se adapta à região e período — risco muito alto." }; if (ehForte && clima.favoravel) return { status: "FAVORÁVEL", mensagem: "Cultura dominante na região, clima compatível — alta probabilidade de sucesso." }; if (ehComplementar && clima.favoravel) return { status: "ACEITÁVEL", mensagem: "Cultura complementar, requer atenção aos tratos." }; if (ehForte && clima.alerta) return { status: "ALERTA", mensagem: "Cultura forte na região, mas com riscos climáticos — veja orientações." }; }
